// Firebase Security Rules for TGDP Ecosystem
// Copy to Firebase Console > Firestore > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return request.auth.uid == userId;
    }
    
    function hasRole(role) {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.roles.hasAny([role]);
    }
    
    function isAdmin() {
      return hasRole('admin');
    }
    
    function isOmbudsman() {
      return hasRole('ombudsman');
    }
    
    // Users collection
    match /users/{userId} {
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow create: if isAuthenticated() && isOwner(userId);
      allow update: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow delete: if isAdmin();
    }
    
    // KYC collection
    match /kyc/{kycId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // TGDP Balances
    match /tgdp_balances/{userId} {
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow write: if false; // Only through Cloud Functions
    }
    
    // TGDP Transactions
    match /tgdp_transactions/{txId} {
      allow read: if isAuthenticated() && (
        resource.data.fromUserId == request.auth.uid ||
        resource.data.toUserId == request.auth.uid ||
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
      allow write: if false; // Only through Cloud Functions
    }
    
    // FTR Balances
    match /ftr_balances/{userId} {
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow write: if false; // Only through Cloud Functions
    }
    
    // GIC Balances
    match /gic_balances/{licenseeId} {
      allow read: if isAuthenticated() && (isOwner(licenseeId) || isAdmin());
      allow write: if false; // Only through Cloud Functions
    }
    
    // Household-Licensee Links
    match /household_links/{linkId} {
      allow read: if isAuthenticated() && (
        resource.data.householdId == request.auth.uid ||
        resource.data.licenseeId == request.auth.uid ||
        isAdmin()
      );
      allow write: if false; // Only through Cloud Functions
    }
    
    // Complaints
    match /complaints/{complaintId} {
      allow read: if isAuthenticated() && (
        resource.data.complainantId == request.auth.uid ||
        isOmbudsman() ||
        isAdmin()
      );
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && (isOmbudsman() || isAdmin());
      allow delete: if isAdmin();
    }
    
    // T-JR Returns
    match /tjr_returns/{returnId} {
      allow read: if isAuthenticated() && (
        resource.data.userId == request.auth.uid ||
        resource.data.jewelerId == request.auth.uid ||
        isAdmin()
      );
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && (
        resource.data.userId == request.auth.uid ||
        hasRole('jeweler') ||
        isAdmin()
      );
    }
    
    // T-JDB Designs
    match /tjdb_designs/{designId} {
      allow read: if true; // Public read for marketplace
      allow create: if isAuthenticated() && hasRole('designer');
      allow update: if isAuthenticated() && (
        resource.data.designerId == request.auth.uid ||
        isAdmin()
      );
      allow delete: if isAdmin();
    }
    
    // Jewelers
    match /jewelers/{jewelerId} {
      allow read: if true; // Public read for selection
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && (isOwner(jewelerId) || isAdmin());
      allow delete: if isAdmin();
    }
    
    // Trading Orders
    match /trading_orders/{orderId} {
      allow read: if isAuthenticated() && (
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Audit Logs (Admin only)
    match /audit_logs/{logId} {
      allow read: if isAdmin();
      allow write: if false; // Only through Cloud Functions
    }
    
    // Configuration (Admin only)
    match /config/{configId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }
  }
}
